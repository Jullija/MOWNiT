function dotVector(vecSize)
    ans = [];
    
    for i in 1:10
        vec1 = rand(vecSize);
        vec2 = rand(vecSize)
        push!(ans, @elapsed dot(vec1, vec2))
    end
    
    return ans
        
end

function matrixVector(size)    
    ans = [];
    
    for i in 1:10
        matrix = rand(size, size);
        vector = rand(size);
        push!(ans, @elapsed *(matrix, vector))
    end
    
    
    return ans
end
